# Comment with %>%
# Only 'x = ' in 'x = 1:10' should be highlighted
tibble(x = 1:10, y =x + 10) %>%
  mutate(z =x + y) %>%
  select(x, z) %>%
  arrange_all() %>%
  filter(!(x == 0)) %>%
  group_by(x %% 3) %>%
  summarise_all(funs(mean(.))) %>%
  summarize(nrow=nrow(.))
